# =============================================================================
# Installation notes:
#
# 1. Create environment:
#    conda create -n aquaagent python=3.10 && conda activate aquaagent
#
# 2. Install PyTorch with correct CUDA version:
#    pip install torch==2.2.0 torchvision==0.17.0 --index-url https://download.pytorch.org/whl/cu121
#
# 3. Install PyTorch Geometric wheels:
#    pip install torch-scatter torch-sparse -f https://data.pyg.org/whl/torch-2.2.0+cu121.html
#    pip install torch-geometric==2.5.0
#
# 4. Install remaining dependencies:
#    pip install -r requirements.txt
#
# NOTE: epyt requires EPANET 2.2 shared libraries. On Linux/macOS these are
#       bundled with epyt. On Windows, install EPANET from
#       https://www.epa.gov/water-research/epanet before installing epyt.
